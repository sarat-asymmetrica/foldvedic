// Calculate confidence using quaternion similarity
fn calculate_confidence(
    flow: &Flow,
    user_profile: &QuaternionProfile,
    context_relevance: f64
) -> f64 {
    // Flow's quaternion signature
    let flow_q = Quaternion::new(
        flow.historical_success,  // w: Past performance
        flow.complexity_inverse,  // x: Simpler = higher confidence
        flow.data_completeness,   // y: More data = more confident
        user_profile.familiarity  // z: User's past choices
    );

    // User's quaternion profile
    let user_q = Quaternion::new(
        user_profile.w,
        user_profile.x,
        user_profile.y,
        user_profile.z
    );

    // Semantic similarity = confidence alignment
    let alignment = flow_q.semantic_similarity(&user_q);

    // Combine with context relevance
    let confidence = (alignment * 0.7) + (context_relevance * 0.3);

    // Clamp to [0, 1]
    confidence.max(0.0).min(1.0)
}

// Classify regime based on confidence + complexity
fn classify_regime(flow: &Flow, confidence: f64) -> Regime {
    if confidence < 0.75 {
        Regime::Exploration  // Low confidence â†’ explore options

        Regime::Optimization // High confidence + low complexity â†’ optimize for speed
    } else {
        Regime::Stabilization // High confidence + moderate complexity â†’ reliable
    }
}
