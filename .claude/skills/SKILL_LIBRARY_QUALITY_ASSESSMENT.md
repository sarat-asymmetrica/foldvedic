## Unified Quality Score (Harmonic Mean)

let scores = [

];

fn harmonic_mean(scores: &[f64]) -> f64 {
    let n = scores.len() as f64;
    let sum_reciprocals: f64 = scores.iter()
        .map(|s| 1.0 / s)
        .sum();
    n / sum_reciprocals
}
